# 如何修复输出值中的 undefined ？(阿里)

```js
function LateBloomer(){
    this.petalCount = Math.ceil(Math.random()*2 + 1);
}
LateBloomer.prototype.bloom = function(){
    window.setTimeout(this.declare, 1000);
}
LateBloomer.prototype.declare = function(){
    console.log('I am a beautiful flower with ' + this.petalCount + ' petals');
}
var flower = new LateBloomer();
flower.bloom();
```
> 结果： I am a beautiful flower with undefined petals

**那么怎么修复输出值中的undefined？**
```js
function LateBloomer(){
    this.petalCount = Math.ceil(Math.random()*2 + 1);
}
LateBloomer.prototype.bloom = function(){
var that = this;
    window.setTimeout(function(){that.declare(that.petalCount)}, 1000);
}
LateBloomer.prototype.declare = function(petalCount){
    console.log('I am a beautiful flower with ' + this.petalCount + ' petals');
}
var flower = new LateBloomer();
flower.bloom();
```
